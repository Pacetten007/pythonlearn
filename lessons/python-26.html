def word_frequency(filename, top_n=10):
    """–ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ—Ç —á–∞—Å—Ç–æ—Ç—É —Å–ª–æ–≤ –≤ —Ñ–∞–π–ª–µ"""
    word_count = {}

    try:
        with open(filename, "r", encoding="utf-8") as file:
            for line in file:
                # –†–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ —Å–ª–æ–≤–∞ –∏ –æ—á–∏—â–∞–µ–º –æ—Ç –∑–Ω–∞–∫–æ–≤ –ø—Ä–µ–ø–∏–Ω–∞–Ω–∏—è

                for word in words:
                    # –£–±–∏—Ä–∞–µ–º –∑–Ω–∞–∫–∏ –ø—Ä–µ–ø–∏–Ω–∞–Ω–∏—è
                    clean_word = word.strip(".,!?;:()[]\"'")

                        # –£–≤–µ–ª–∏—á–∏–≤–∞–µ–º —Å—á–µ—Ç—á–∏–∫
                        word_count[clean_word] = word_count.get(clean_word, 0) + 1

        sorted_words = sorted(word_count.items(), key=lambda x: x[1], reverse=True)

        # –í—ã–≤–æ–¥–∏–º —Ç–æ–ø-N —Å–ª–æ–≤
        print(f"–¢–æ–ø-{top_n} —Å–∞–º—ã—Ö —á–∞—Å—Ç—ã—Ö —Å–ª–æ–≤:")
        for i, (word, count) in enumerate(sorted_words[:top_n], 1):
            print(f"{i}. '{word}' - {count} —Ä–∞–∑")

    except FileNotFoundError:
        print(f"–§–∞–π–ª '{filename}' –Ω–µ –Ω–∞–π–¥–µ–Ω!")

    """–ê–Ω–∞–ª–∏–∑–∏—Ä—É–µ—Ç –ª–æ–≥-—Ñ–∞–π–ª –∏ –≤—ã–≤–æ–¥–∏—Ç —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É"""
    stats = {

        'ERROR': 0,

    }
    errors = []

    try:
        with open(filename, "r", encoding="utf-8") as file:

                # –ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ–º —Ç–∏–ø—ã —Å–æ–æ–±—â–µ–Ω–∏–π

        print("=== –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ª–æ–≥–æ–≤ ===")

        if errors:
            print("\n=== –ù–∞–π–¥–µ–Ω–Ω—ã–µ –æ—à–∏–±–∫–∏ ===")
            for line_num, line in errors:
                print(f"–°—Ç—Ä–æ–∫–∞ {line_num}: {line}")

        print(f"–§–∞–π–ª '{filename}' –Ω–µ –Ω–∞–π–¥–µ–Ω!")
